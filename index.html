* Обновите курсы перед поездкой */ const CURRENCIES = { 'EUR': { symbol: '€', rate: 1, keywords: ['евро', 'евр', 'euro'] }, 'TRY': { symbol: '₺', rate: 0.028, keywords: ['лир', 'лира', 'лиры', 'try'] }, 'RSD': { symbol: 'дин', rate: 0.0085, keywords: ['динар', 'динаров', 'дин', 'rsd'] }, 'BGN': { symbol: 'лв', rate: 0.51, keywords: ['лев', 'лева', 'левов', 'лв', 'bgn'] } }; let entries = JSON.parse(localStorage.getItem('triplog_entries') || '[]'); // --- ИНИЦИАЛИЗАЦИЯ --- function init() { render(); setupVoice(); } // --- ГОЛОСОВОЕ УПРАВЛЕНИЕ --- function setupVoice() { const micBtn = document.getElementById('micBtn'); const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { micBtn.style.display = 'none'; return; } const recognition = new SpeechRecognition(); recognition.lang = 'ru-RU'; recognition.interimResults = false; micBtn.onclick = () => { recognition.start(); micBtn.classList.add('recording'); document.getElementById('micLabel').innerText = 'СЛУШАЮ...'; }; recognition.onresult = (event) => { const text = event.results[0][0].transcript.toLowerCase(); parseEntry(text); }; recognition.onend = () => { micBtn.classList.remove('recording'); document.getElementById('micLabel').innerText = 'ЗАПИСАТЬ ГОЛОСОМ'; }; } // --- ПАРСИНГ ТЕКСТА --- function parseEntry(text) { // Пример: "Бензин 500 лир" или "Обед 20 евро" const amountMatch = text.match(/\d+/); if (!amountMatch) { showToast("Не услышал сумму"); return; } const amount = parseFloat(amountMatch[0]); let category = 'other'; let currency = document.getElementById('currencySelect').value; // Определяем категорию if (text.includes('бензин') || text.includes('топливо') || text.includes('газ')) category = 'fuel'; if (text.includes('отель') || text.includes('жилье') || text.includes('ночлег')) category = 'hotel'; if (text.includes('еда') || text.includes('обед') || text.includes('ужин') || text.includes('ресторан')) category = 'food'; // ОПРЕДЕЛЕНИЕ ВАЛЮТЫ ИЗ ГОЛОСА for (const [code, info] of Object.entries(CURRENCIES)) { if (info.keywords.some(kw => text.includes(kw))) { currency = code; break; } } const newEntry = { id: Date.now(), title: text.replace(/\d+/g, '').replace(/(евро|лир|динар|лев|бензин|отель|еда|обед|ужин)/gi, '').trim() || 'Расход', amount: amount, currency: currency, category: category, date: new Date().toISOString() }; entries.unshift(newEntry); save(); showToast(`Записано: ${amount} ${CURRENCIES[currency].symbol}`); } // --- СОХРАНЕНИЕ И РЕНДЕР --- function save() { localStorage.setItem('triplog_entries', JSON.stringify(entries)); render(); } function render() { const historyCont = document.getElementById('history'); const heroTotal = document.getElementById('heroTotal'); const filterCat = document.getElementById('filterCat').value; historyCont.innerHTML = ''; let totalEur = 0; entries.forEach(entry => { const rate = CURRENCIES[entry.currency]?.rate || 1; const inEur = entry.amount * rate; totalEur += inEur; if (filterCat && entry.category !== filterCat) return; const div = document.createElement('div'); div.className = 'entry'; div.innerHTML = `
${getIcon(entry.category)}
${entry.title}
${new Date(entry.date).toLocaleTimeString()}
${entry.amount} ${CURRENCIES[entry.currency].symbol}
`; historyCont.appendChild(div); }); const parts = totalEur.toFixed(2).split('.'); heroTotal.innerHTML = `${parts[0]},${parts[1]} €`; } function getIcon(cat) { const icons = { fuel: '⛽', hotel: '🏨', food: '🍕', other: '📦' }; return icons[cat] || '💰'; } function deleteEntry(id) { entries = entries.filter(e => e.id !== id); save(); } function showToast(msg) { const t = document.getElementById('toast'); t.innerText = msg; t.classList.add('show'); setTimeout(() => t.classList.remove('show'), 3000); } document.getElementById('filterCat').onchange = render; init();
